import numpy as np
import matplotlib.pyplot as plt

H, l, p = 18, 64, 0.001
U = np.arange(50, 2000, 10)
L = H + U
E = U * (1-p)**(8*L) / (U + H + l)

plt.plot(U, E)
plt.xlabel('U (–±–∞–π—Ç)')
plt.ylabel('E')
plt.grid()
plt.savefig('E_vs_U.png')
plt.show()

U_opt = U[np.argmax(E)]
print(f'U_opt = {U_opt}, E_max = {np.max(E):.3f}')
